#include "main.h"
/**
 * print_digits - print each digit in a number
 * @tmp: integer value
 *
 * Returns: Always 0
 */
void print_digits(int tmp)
{
	int unit, tens, huns;

	if (tmp > 99)
	{
		unit = tmp % 10;
		tens = ((tmp - unit) % 100) / 10;
		huns = (tmp - tens * 10 - unit) / 100;
		_putchar('0' + huns);
		_putchar('0' + tens);
		_putchar('0' + unit);
	}
	else
	{
		if (tmp > 9)
		{
			unit = tmp % 10;
			tens = (tmp - unit) / 10;
			_putchar('0' + tens);
			_putchar('0' + unit);
		}
		else
		{
			_putchar('0' + tmp);
		}
	}
}
